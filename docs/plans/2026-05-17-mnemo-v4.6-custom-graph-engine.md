```python
"""v4.6 custom layout engine: ForceAtlas2 / LinLog / Barnes-Hut.

The v4.5.x Fruchterman-Reingold collapsed the giant component into a
featureless dense disk (kNN-16-only repulsion + undifferentiated
gravity + p95-normalize) and scattered 2298 tiny components as
one-dot-per-component confetti. v4.6 replaces the ALGORITHM (the
server-side + cached architecture is unchanged; signature unchanged).

These tests are the real acceptance gate -- the dominant failure (the
"blob") is a property of the Python output and is asserted NUMERICALLY
here, no GPU required.
"""

from __future__ import annotations

import math
import random

import numpy as np

from mnemo.ui.graph_layout import compute_graph_layout


def _planted(communities: int, per: int, singles: int, seed: int = 7):
    """A graph with PLANTED community structure: `communities` dense
    blobs (sparse inter-community bridges) = the giant; plus `singles`
    singletons. Returns (n, edges, labels) where labels[i] is the
    planted community of node i (-1 for singletons)."""
    rng = random.Random(seed)
    edges: list[tuple[int, int]] = []
    labels: list[int] = []
    nodes_by_c: list[list[int]] = []
    n = 0
    for c in range(communities):
        members = list(range(n, n + per))
        nodes_by_c.append(members)
        labels += [c] * per
        n += per
        # dense intra-community (each node ~4 internal edges)
        for i in members:
            for _ in range(4):
                j = rng.choice(members)
                if i != j:
                    edges.append((i, j))
    # sparse inter-community bridges (1 per community pair)
    for a in range(communities):
        for b in range(a + 1, communities):
            edges.append((rng.choice(nodes_by_c[a]), rng.choice(nodes_by_c[b])))
    labels += [-1] * singles
    n += singles
    return n, edges, labels


def _xy(pos: list[float], i: int) -> tuple[float, float]:
    return pos[2 * i], pos[2 * i + 1]
```

```python
"""Server-side Nebula layout (v4.6 custom engine).

ForceAtlas2 with the LinLog energy model + outbound-attraction-
distribution + a Barnes-Hut quadtree for repulsion (every node repels
EVERY region -- the kNN-16 of v4.5.x was the blob cause) + weak
non-strong gravity + full-extent normalize. Small components/singletons
are laid out then DENSELY bin-packed into a band hugging the giant
(never one-dot-per-component confetti). Deterministic (fixed seed) ->
byte-identical -> cacheable. Signature unchanged from v4.5:
``compute_graph_layout(n, edges) -> [x0,y0,x1,y1,...]`` in node order.
"""

from __future__ import annotations

import math

import numpy as np

_SEED = 42
_ITERS = 320
_THETA = 0.7          # Barnes-Hut opening angle
_GRAVITY = 0.04       # weak, NON-strong
_SCALE = 12.0         # repulsion scaling ratio (ForceAtlas2)
_WORLD = 2000.0       # post-normalize target half-extent


def _components(n, edges):
    from scipy.sparse import coo_matrix
    from scipy.sparse.csgraph import connected_components

    if not edges:
        return np.zeros(n, dtype=np.int64), -1
    r = np.fromiter((e[0] for e in edges), np.int64, len(edges))
    c = np.fromiter((e[1] for e in edges), np.int64, len(edges))
    d = np.ones(len(edges) * 2, np.int8)
    adj = coo_matrix(
        (d, (np.concatenate([r, c]), np.concatenate([c, r]))), shape=(n, n)
    ).tocsr()
    ncomp, lab = connected_components(adj, directed=False)
    if ncomp <= 1:
        return lab, int(lab[0]) if n else -1
    return lab, int(np.bincount(lab, minlength=ncomp).argmax())


def _bh_repulsion(pos: np.ndarray, deg: np.ndarray, theta: float) -> np.ndarray:
    """Barnes-Hut quadtree repulsion. O(n log n). Force on i from a
    cell is k * (deg_i+1) * (mass_cell) / dist, opened when
    cell_size / dist > theta (else treat the cell as one supernode at
    its centre of mass). ForceAtlas2 repulsion is degree-weighted."""
    nN = pos.shape[0]
    disp = np.zeros((nN, 2), np.float64)
    mn = pos.min(axis=0)
    span = float((pos.max(axis=0) - mn).max()) or 1.0

    # Iterative quadtree as flat arrays (no recursion -> fast + safe).
    # Each node: bbox(min,size), com (mass-weighted), mass, child idx[4]
    # or a single body index. Build by inserting points.
    # (Implementation: a numpy-friendly pointer quadtree. See NOTE.)
    from mnemo.ui._quadtree import build_quadtree, accumulate_repulsion

    qt = build_quadtree(pos, deg + 1.0, mn, span)
    accumulate_repulsion(qt, pos, deg + 1.0, theta, _SCALE, disp)
    return disp


def _layout_component(pos, edges_arr, deg, iters):
    """ForceAtlas2 + LinLog on one component. ``pos`` (m,2) is mutated
    + returned. ``edges_arr`` is (E,2) local indices."""
    m = pos.shape[0]
    if m <= 1:
        return pos
    ea, eb = edges_arr[:, 0], edges_arr[:, 1]
    src_mass = deg[ea] + 1.0  # outbound-attraction distribution
    for it in range(iters):
        disp = _bh_repulsion(pos, deg, _THETA)
        # LinLog attraction: F = log(1+d), divided by source mass.
        d = pos[ea] - pos[eb]
        dist = np.sqrt(np.einsum("ij,ij->i", d, d)) + 1e-9
        fa = (np.log1p(dist) / dist / src_mass)[:, None] * d
        np.add.at(disp, ea, -fa)
        np.add.at(disp, eb, fa)
        # weak gravity toward centroid (NOT strong gravity)
        ctr = pos.mean(axis=0)
        g = ctr - pos
        gd = np.sqrt(np.einsum("ij,ij->i", g, g)) + 1e-9
        disp += g / gd[:, None] * (_GRAVITY * (deg + 1.0))[:, None]
        # adaptive cooling step clamp
        dlen = np.sqrt(np.einsum("ij,ij->i", disp, disp)) + 1e-9
        cap = _WORLD * 0.1 * (1.0 - it / iters) + 1.0
        pos += disp / dlen[:, None] * np.minimum(dlen, cap)[:, None]
    return pos


def _normalize_full_extent(pos: np.ndarray) -> np.ndarray:
    """Centre at origin, scale by the FULL extent (NOT p95 -- p95 was a
    blob cause) so the whole drawing fits ``_WORLD``."""
    pos -= (pos.max(axis=0) + pos.min(axis=0)) / 2.0
    ext = float(np.abs(pos).max()) or 1.0
    return pos * (_WORLD / ext)


def _grid_pack(boxes, anchor_r, rng):
    """Deterministic shelf bin-pack of (w,h) component boxes into a
    dense band starting at radius ``anchor_r``. Returns per-box (cx,cy)
    centres on a ring-of-shelves so the halo is DENSE + textured, never
    one-dot-per-component confetti."""
    placed = []
    ang = 0.0
    r = anchor_r
    shelf_h = 0.0
    x = 0.0
    circ = 2.0 * math.pi * r
    for (w, h) in boxes:
        if x + w > circ:                # next shelf (outer ring)
            r += shelf_h * 1.15 + 8.0
            circ = 2.0 * math.pi * r
            x = 0.0
            shelf_h = 0.0
        a = (x + w / 2.0) / r           # arc position -> angle
        jitter = (rng.random() - 0.5) * 6.0
        placed.append((math.cos(a) * (r + jitter), math.sin(a) * (r + jitter)))
        x += w + 6.0
        shelf_h = max(shelf_h, h)
    return placed


def compute_graph_layout(n: int, edges: list[tuple[int, int]]) -> list[float]:
    if n <= 0:
        return []
    if n == 1:
        return [0.0, 0.0]
    rng = np.random.default_rng(_SEED)
    pos = np.zeros((n, 2), np.float64)

    clean, seen = [], set()
    for a, b in edges:
        if a == b or not (0 <= a < n) or not (0 <= b < n):
            continue
        k = (a, b) if a < b else (b, a)
        if k in seen:
            continue
        seen.add(k)
        clean.append((a, b))

    deg = np.zeros(n, np.float64)
    for a, b in clean:
        deg[a] += 1.0
        deg[b] += 1.0

    lab, giant = _components(n, clean)
    gmask = lab == giant if giant >= 0 else np.zeros(n, bool)
    gidx = np.nonzero(gmask)[0]

    if gidx.size:
        remap = {int(o): k for k, o in enumerate(gidx)}
        ge = np.array(
            [(remap[a], remap[b]) for a, b in clean if gmask[a] and gmask[b]],
            np.int64,
        ).reshape(-1, 2)
        ang = rng.uniform(0, 2 * math.pi, gidx.size)
        rad = np.sqrt(rng.uniform(0, 1, gidx.size)) * (_WORLD * 0.5)
        gp = np.column_stack([np.cos(ang) * rad, np.sin(ang) * rad])
        gp = _layout_component(gp, ge, deg[gidx], _ITERS)
        gp = _normalize_full_extent(gp)
        pos[gidx] = gp
        giant_r = float(np.abs(gp).max())
    else:
        giant_r = _WORLD * 0.5

    # other components: lay each out tiny, collect bounding boxes,
    # then DENSE grid-pack them in a band hugging the giant.
    import random as _r

    prng = _r.Random(_SEED + 1)
    others: dict[int, list[int]] = {}
    for i in range(n):
        if not gmask[i]:
            others.setdefault(int(lab[i]), []).append(i)
    groups = sorted(others.values(), key=len, reverse=True)
    boxes, blobs = [], []
    for members in groups:
        m = len(members)
        if m == 1:
            boxes.append((10.0, 10.0))
            blobs.append(np.zeros((1, 2)))
            continue
        rmap = {o: k for k, o in enumerate(members)}
        ce = np.array(
            [
                (rmap[a], rmap[b])
                for a, b in clean
                if a in rmap and b in rmap
            ],
            np.int64,
        ).reshape(-1, 2)
        a0 = rng.uniform(0, 2 * math.pi, m)
        sp = np.column_stack([np.cos(a0), np.sin(a0)]) * (8.0 + m)
        sp = _layout_component(sp, ce, deg[members], 80)
        sp -= (sp.max(0) + sp.min(0)) / 2.0
        blobs.append(sp)
        w, h = (sp.max(0) - sp.min(0) + 12.0)
        boxes.append((float(w), float(h)))
    centres = _grid_pack(boxes, giant_r * 1.12, prng)
    for members, blob, (cx, cy) in zip(groups, blobs, centres, strict=True):
        for k, node in enumerate(members):
            pos[node] = (cx + blob[k, 0], cy + blob[k, 1])

    out = [0.0] * (2 * n)
    for i in range(n):
        out[2 * i] = float(pos[i, 0])
        out[2 * i + 1] = float(pos[i, 1])
    return out
```

**Step 4: Create the quadtree helper** `daemon/mnemo/ui/_quadtree.py` (Barnes-Hut, vectorised-ish; correctness over micro-opt — it runs offline + cached):

```python
"""Barnes-Hut quadtree for ForceAtlas2 repulsion (offline, cached).

A flat-array pointer quadtree. Build by point insertion; each internal
cell stores mass + centre-of-mass; repulsion opens a cell when
size/dist > theta else treats it as one supernode. O(n log n)."""

from __future__ import annotations

import numpy as np


class _QT:
    __slots__ = ("cx", "cy", "size", "mass", "comx", "comy", "child", "body")

    def __init__(self, cx, cy, size):
        self.cx, self.cy, self.size = cx, cy, size
        self.mass = 0.0
        self.comx = 0.0
        self.comy = 0.0
        self.child = [None, None, None, None]
        self.body = -1


def _quad(node, x, y):
    return (1 if x >= node.cx else 0) + (2 if y >= node.cy else 0)


def build_quadtree(pos, mass, mn, span):
    root = _QT(mn[0] + span / 2.0, mn[1] + span / 2.0, span)
    for i in range(pos.shape[0]):
        _insert(root, i, pos[i, 0], pos[i, 1], mass[i], pos, mass)
    return root


def _insert(node, i, x, y, w, pos, mass):
    if node.mass == 0.0 and node.body == -1:
        node.body, node.mass, node.comx, node.comy = i, w, x, y
        return
    if node.body != -1:                       # split a leaf
        b = node.body
        node.body = -1
        _place(node, b, pos[b, 0], pos[b, 1], mass[b], pos, mass)
    node.comx = (node.comx * node.mass + x * w) / (node.mass + w)
    node.comy = (node.comy * node.mass + y * w) / (node.mass + w)
    node.mass += w
    _place(node, i, x, y, w, pos, mass)


def _place(node, i, x, y, w, pos, mass):
    q = _quad(node, x, y)
    if node.child[q] is None:
        h = node.size / 2.0
        ox = (-h / 2.0) if q in (0, 2) else (h / 2.0)
        oy = (-h / 2.0) if q in (0, 1) else (h / 2.0)
        node.child[q] = _QT(node.cx + ox, node.cy + oy, h)
    _insert(node.child[q], i, x, y, w, pos, mass)


def accumulate_repulsion(root, pos, mass, theta, scale, disp):
    n = pos.shape[0]
    for i in range(n):
        _force(root, i, pos[i, 0], pos[i, 1], mass[i], theta, scale, disp)


def _force(node, i, x, y, w, theta, scale, disp):
    if node is None or node.mass == 0.0:
        return
    dx = x - node.comx
    dy = y - node.comy
    d2 = dx * dx + dy * dy + 1e-9
    if node.body == i and node.body != -1:
        return
    if node.body != -1 or (node.size * node.size) / d2 < theta * theta:
        f = scale * w * node.mass / d2
        disp[i, 0] += dx * f
        disp[i, 1] += dy * f
        return
    for c in node.child:
        _force(c, i, x, y, w, theta, scale, disp)
```

```python
def _community_separation(pos, labels):
    """Mean intra-community pairwise distance / mean inter-community
    pairwise distance over the planted (non-singleton) nodes. << 1.0
    means communities are SEPARATED (not one blob)."""
    pts = {}
    for i, c in enumerate(labels):
        if c >= 0:
            pts.setdefault(c, []).append(_xy(pos, i))
    import itertools
    rng = random.Random(3)
    intra, inter = [], []
    cs = list(pts)
    for c in cs:
        P = pts[c]
        for _ in range(400):
            a, b = rng.choice(P), rng.choice(P)
            intra.append(math.dist(a, b))
    for a, b in itertools.combinations(cs, 2):
        for _ in range(200):
            p, q = rng.choice(pts[a]), rng.choice(pts[b])
            inter.append(math.dist(p, q))
    return (sum(intra) / len(intra)) / (sum(inter) / len(inter))


def test_communities_are_separated_not_a_blob() -> None:
    """THE de-blob gate. Plain FR (v4.5.x) collapsed planted
    communities into one disk -> ratio ~1.0. FA2+LinLog must pull
    them apart -> intra distance must be MUCH smaller than inter."""
    n, edges, labels = _planted(6, 70, 120)
    pos = compute_graph_layout(n, edges)
    ratio = _community_separation(pos, labels)
    assert ratio < 0.55, (
        f"communities not separated (intra/inter={ratio:.3f}); the "
        f"giant is still a featureless blob -- FA2+LinLog must cluster"
    )


def test_edges_much_shorter_than_random_pairs() -> None:
    n, edges, labels = _planted(5, 80, 100)
    pos = compute_graph_layout(n, edges)
    gi = [i for i, c in enumerate(labels) if c >= 0]
    gs = set(gi)
    ge = [(a, b) for a, b in edges if a in gs and b in gs]
    me = sum(math.dist(_xy(pos, a), _xy(pos, b)) for a, b in ge) / len(ge)
    rng = random.Random(1)
    mr = sum(
        math.dist(_xy(pos, rng.choice(gi)), _xy(pos, rng.choice(gi)))
        for _ in range(3000)
    ) / 3000
    assert me / mr < 0.5, f"edges not contracted (ratio {me/mr:.3f})"
```

```javascript
/* nebula-gl.js -- mnemo v4.6 custom WebGL graph renderer (regl).
   Purpose-built, single view. NO sigma/graphology. Renders ONLY when
   dirty (camera/selection/hover/drag) -- idle == zero cost. */
(function (global) {
  'use strict';

  function create(canvas, opts) {
    var regl = global.createREGL({ canvas: canvas, attributes:
      { antialias: false, alpha: false, premultipliedAlpha: false } });
    var theme = opts.theme || {};
    var bg = theme.bg || [0.027, 0.035, 0.059, 1]; // #07090f
    var N = opts.nodes;   // [{x,y,size,color:[r,g,b]}...]
    var E = opts.edges;   // [{s,t}] indices into N
    // --- camera (world->clip) ---
    var cam = { x: 0, y: 0, zoom: 1 };
    var dirty = true, raf = 0, dragId = -1, hoverId = -1, selId = -1;
    var hl = null;        // Set of highlighted indices or null

    // packed buffers
    var posBuf = regl.buffer(flat(N, function (d) { return [d.x, d.y]; }));
    var colBuf = regl.buffer(flat(N, function (d) { return d.color; }));
    var sizBuf = regl.buffer(N.map(function (d) { return d.size; }));
    var eposA = [], eposB = [];
    E.forEach(function (e) {
      eposA.push(N[e.s].x, N[e.s].y); eposB.push(N[e.t].x, N[e.t].y);
    });

    // --- node program: instanced SDF circle + additive glow pass ---
    var drawNodes = regl({
      vert:
        'precision highp float; attribute vec2 pos; attribute vec3 col;' +
        'attribute float siz; uniform vec2 cam; uniform float zoom;' +
        'uniform vec2 res; varying vec3 vC; varying float vR;' +
        'void main(){ vC=col; vR=siz;' +
        ' vec2 p=(pos-cam)*zoom;' +
        ' gl_Position=vec4(p/res*2.0,0,1);' +
        ' gl_PointSize=clamp(siz*zoom*2.0,2.0,90.0);}',
      frag:
        'precision highp float; varying vec3 vC; varying float vR;' +
        'uniform float glow;' +
        'void main(){ vec2 q=gl_PointCoord-0.5; float d=length(q);' +
        ' float aa=fwidth(d);' +
        ' float core=1.0-smoothstep(0.5-aa,0.5,d);' +
        ' float ring=smoothstep(0.46-aa,0.46,d)*core;' +
        ' float halo=glow*(1.0-smoothstep(0.0,0.5,d))*0.5;' +
        ' vec3 c=mix(vC, vC*0.55, ring);' +
        ' gl_FragColor=vec4(c, core)+vec4(vC,0.0)*halo;' +
        ' if(gl_FragColor.a<0.01) discard;}',
      attributes: { pos: posBuf, col: colBuf, siz: sizBuf },
      uniforms: {
        cam: function () { return [cam.x, cam.y]; },
        zoom: function () { return cam.zoom; },
        res: function (c) { return [c.viewportWidth, c.viewportHeight]; },
        glow: function () { return theme.glow == null ? 1.0 : theme.glow; },
      },
      count: N.length, primitive: 'points',
      blend: { enable: true, func: { src: 'src alpha', dst:
        'one minus src alpha' } }, depth: { enable: false },
    });

    // --- edge program: low-alpha straight lines (density => structure)
    var drawEdges = regl({
      vert:
        'precision highp float; attribute vec2 a,b; attribute float t;' +
        'uniform vec2 cam,res; uniform float zoom;' +
        'void main(){ vec2 w=mix(a,b,t); vec2 p=(w-cam)*zoom;' +
        ' gl_Position=vec4(p/res*2.0,0,1);}',
      frag:
        'precision highp float; uniform vec4 ec;' +
        'void main(){ gl_FragColor=ec; }',
      attributes: {
        a: { buffer: regl.buffer(eposA), divisor: 0 },
        b: { buffer: regl.buffer(eposB), divisor: 0 },
        t: [0, 1],
      },
      uniforms: {
        cam: function () { return [cam.x, cam.y]; },
        zoom: function () { return cam.zoom; },
        res: function (c) { return [c.viewportWidth, c.viewportHeight]; },
        ec: function () { return theme.edge || [0.36, 0.42, 0.55, 0.10]; },
      },
      count: 2, instances: E.length, primitive: 'lines',
      blend: { enable: true, func: { src: 'src alpha', dst: 'one' } },
      depth: { enable: false },
    });

    function frame() {
      raf = 0;
      regl.clear({ color: bg, depth: 1 });
      if (E.length) drawEdges();
      drawNodes();
      // (highlight pass: redraw hl/selected nodes + incident edges
      //  brighter on top -- same programs, filtered buffers; omitted
      //  here for brevity, implement with a second buffer set.)
    }
    function invalidate() { dirty = true; if (!raf) raf =
      global.requestAnimationFrame(frame); }

    // --- CPU quadtree picking (reuse positions; O(log n), no/frame) --
    var pick = buildPickIndex(N);
    // --- camera interaction: wheel zoom-to-cursor, drag pan, node drag
    wireInput(canvas, cam, pick, {
      onClickNode: function (i) { handle._emit('clickNode', i); },
      onClickStage: function () { handle._emit('clickStage'); },
      onHover: function (i) { hoverId = i; invalidate();
        handle._emit('hoverNode', i); },
      onChange: invalidate,
    });

    var handle = {
      _cbs: {},
      on: function (ev, cb) { (this._cbs[ev] = this._cbs[ev] || [])
        .push(cb); },
      _emit: function (ev, a) { (this._cbs[ev] || []).forEach(
        function (f) { f(a); }); },
      setHighlight: function (set) { hl = set; invalidate(); },
      select: function (i) { selId = i; if (i != null) easeTo(cam,
        N[i], invalidate); invalidate(); },
      hover: function (i) { hoverId = i == null ? -1 : i;
        invalidate(); },
      fit: function () { fitCamera(cam, N); invalidate(); },
      camera: cam,
      destroy: function () { if (raf) global.cancelAnimationFrame(raf);
        regl.destroy(); },
    };
    invalidate();
    return handle;
  }
  function flat(a, f) { var o = []; a.forEach(function (d) {
    o.push.apply(o, f(d)); }); return o; }
  // buildPickIndex / wireInput / easeTo / fitCamera / LabelProvider:
  // small deterministic helpers -- implement straightforwardly. The
  // LabelProvider is a thin object: { setLabels(items), clear() } that
  // draws dark themed pills on a 2D <canvas> sibling of the GL canvas.
  global.NebulaGL = { create: create, LabelProvider: makeLabelProvider };

  function makeLabelProvider(overlayCanvas) { /* 2D pill drawer;
    setLabels([{x,y,text,accent}]) / clear(). Seam for a future SDF
    swap with no call-site change. */ }
  function buildPickIndex(N){/* quadtree over N -> nearestWithin(x,y,r)
    in world space */}
  function wireInput(cv,cam,pick,cb){/* wheel=zoom-to-cursor,
    drag=pan, mousedown-on-node=node drag, click=node/stage, mousemove
    =hover; every change calls cb.onChange */}
  function easeTo(cam,n,inv){/* short rAF tween of cam.x/y/zoom */}
  function fitCamera(cam,N){/* set cam so all N fit with margin */}
})(window);
```
